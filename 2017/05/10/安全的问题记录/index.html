种子表中四个主要的字段的描述

budget_control_name 基本的和tag表保持一致的一个字段，用于扣血和回血

budget_condition 记录bizId

budget_biz_data 记录扣血日期

condition_str 补充标记种子唯一性的字段

select app_dataset_id,substr(name,1,3),avaliable_budget from app_dataset_tag_info where avaliable_budget < 30 order by avaliable_budget;

select substr(name,1,3) from app_dataset_tag_info where avaliable_budget < 30 order by avaliable_budget;

select app_dataset_id,substr(name,1,3),avaliable_budget from app_dataset_tag_info where avaliable_budget < 30 and substr(name,1,3) in (233,254,598) order by avaliable_budget;

select * from app_dataset_tag_info where avaliable_budget < 30 and substr(name,1,3) in (233,254,598) order by avaliable_budget;

select from group_info where smart_id in (200,292,574,574,277,277,277,574,574,574,574,297,574,574,574,279,279,279,574,297,574,574,574,574,574,574,574,277,574,574,574,598,233,540,540,435,297,254,435) and group_type = 0;
select from group_brand where smart_id = 254;

select count(*),budget_biz_date from dp_encrypt_seed where budget_biz_date > “2017-06-15 00:00:00” and budget_biz_date < “2017-06-22 00:00:00” group by budget_biz_date

select
from app_dataset_tag_info t1 where budget - avaliable_budget < (select count() from dp_encrypt_seed t2
where t1.name = t2.budget_control_name and t1.app_dataset_id = t2.dataset_id
and t2.budget_renewed_flag = 0)
order by avaliable_budget

select
from app_dataset_tag_info t1 where budget - avaliable_budget < (select count() as cnt from dp_encrypt_seed t2
where t1.name = t2.budget_control_name and t1.app_dataset_id = t2.dataset_id
and t2.budget_renewed_flag = 0 having cnt > 42)
